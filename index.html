using System;
using System.Collections.Generic;
using System.Text;
                        
namespace PIR_201_Veras_Lab_5
{
    public class Manager : Worker
    {
        private const int h = 100;
                        
        private double sales;
                        
        public double Sales
        {
            get
            {
                return sales;
            }
            set
            {
                sales = value < 0 ? 0 : value;
            }
        }
                        
        public Manager(string name, double minimumSalary, double k, int sales) : base(name, minimumSalary, k)
        {
            this.sales = sales;
        }
                        
        public override double FinalSalary()
        {
        if (sales > h)
        {
            return base.FinalSalary() * h * 0.01;
        }
        else
        {
            return MinimumSalary;
        }
                        
        public override void PrintInfo()
        {
            Console.WriteLine($"Имя сотрудника: {Name}\nДолжность: Менеджер\nМинимальная зарплата: {MinimumSalary}\nОбъём продаж: {Sales}");
        }
    }
}
